```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def distributeCoins(self, root: TreeNode) -> int:
        self.ans = 0 # count the move steps
        
        
        def dfs(TreeNode):
            if not TreeNode: return 0
            L, R = dfs(TreeNode.left), dfs(TreeNode.right)
            self.ans += abs(L) + abs(R) 
            return L + R + TreeNode.val - 1 # node's cc
        
        dfs(root)
        return self.ans

```
